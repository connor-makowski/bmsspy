\begin{algorithm}[!htbp]
\SingleSpacedXI
\caption{{\color{ForestGreen}Reconstruct Path (\texttt{reconstruct\_path})}}
\label{alg:utils_reconstruct_path}
\SetAlgoLined
\KwIn{destination node $dest$, predecessor list $pred$}
\KwOut{list $L$ representing the reconstructed path from origin to $dest$}

$L \leftarrow [\,dest\,]$\;
\While{$pred[dest] \neq -1$}{
  $dest \leftarrow pred[dest]$\;
  append $dest$ to $L$\;
}
\textsc{Reverse}($L$)\;
\Return{$L$}\;
\end{algorithm}
