 10def is_pivot(root: int, forest: dict[int, set[int]], threshold: int) -> bool:
 11    """
 12    Function:
 13
 14    - Returns True if the number of reachable nodes meets or exceeds a given threshold.
 15    - Returns False otherwise.
 16
 17    Required Arguments:
 18
 19    - `root`
 20        - Type: int
 21        - What: The starting node for the DFS traversal.
 22    - `forest`
 23        - Type: dict[int, set[int]]
 24        - What: Adjacency list representing the directed forest.
 25    - `threshold`
 26        - Type: int
 27        - What: The minimum number of reachable nodes required to return True.
 28    """
 29    seen = set()
 30    stack = [root]
 31    cnt = 0
 32    while stack:
 33        x = stack.pop()
 34        if x in seen:
 35            continue
 36        cnt += 1
 37        if cnt >= threshold:
 38            return True
 39        seen.add(x)
 40        stack.extend(forest[x])
 41    return False

169    def find_pivots(
170        self, upper_bound: int | float, frontier: set[int]
171    ) -> tuple[set[int], set[int]]:
172        """
173        Function:
174
175        - Finds pivot sets pivots and temp_frontier according to Algorithm 1.
176
177        Required Arguments:
178
179        - upper_bound:
180            - Type: int | float
181            - What: The upper bound threshold (B)
182        - frontier:
183            - Type: set[int]
184            - What: Set of vertices (S)
185
186        Optional Arguments:
187
188        - None
189
190        Returns:
191
192        - pivots:
193            - Type: Set[int]
194            - What: Set of pivot vertices
195        - frontier:
196            - Type: Set[int]
197            - What: Return a new frontier set of vertices within the upper_bound
198        """
199        temp_frontier = set(frontier)
200        prev_frontier = set(frontier)
201
202        # Multi-step limited relaxation from current frontier
203        for _ in range(self.pivot_relaxation_steps):
204            curr_frontier = set()
205            for prev_frontier_idx in prev_frontier:
206                prev_distance = self.counter_distance_matrix[prev_frontier_idx]
207                for connection_idx, connection_distance in self.graph[
208                    prev_frontier_idx
209                ].items():
210                    # Modification: Use a new get distance function to ensure unique lengths
211                    new_distance = (
212                        prev_distance
213                        + connection_distance
214                        + self.counter_value
215                        + self.edge_adj_graph[prev_frontier_idx][connection_idx]
216                    )
217                    # Important: Allow equality on relaxations
218                    if (
219                        new_distance
220                        <= self.counter_and_edge_distance_matrix[connection_idx]
221                    ):
222                        # Addition: Add predecessor tracking
223                        if (
224                            new_distance
225                            < self.counter_and_edge_distance_matrix[
226                                connection_idx
227                            ]
228                        ):
229                            self.predecessor[connection_idx] = prev_frontier_idx
230                            self.counter_and_edge_distance_matrix[
231                                connection_idx
232                            ] = new_distance
233                            self.counter_distance_matrix[connection_idx] = (
234                                prev_distance
235                                + connection_distance
236                                + self.counter_value
237                            )
238                        if new_distance < upper_bound:
239                            curr_frontier.add(connection_idx)
240            temp_frontier.update(curr_frontier)
241            prev_frontier = curr_frontier
242            # If the search balloons, take the current frontier as pivots
243            if len(temp_frontier) > self.pivot_relaxation_steps * len(frontier):
244                pivots = set(frontier)
245                return pivots, temp_frontier
246
247        # Build tight-edge forest F on temp_frontier: edges (u -> v) with db[u] + w == db[v]
248        forest = {i: set() for i in temp_frontier}
249        indegree = {i: 0 for i in temp_frontier}
250        for frontier_idx in temp_frontier:
251            # prev_distance = self.counter_distance_matrix[frontier_idx]
252            for connection_idx, connection_distance in self.graph[
253                frontier_idx
254            ].items():
255                # Modification: Use predecessor tracking instead of distance comparison
256                if self.predecessor[connection_idx] == frontier_idx:
257                    if connection_idx in temp_frontier:
258                        # direction is frontier_idx -> connection_idx (parent to child)
259                        forest[frontier_idx].add(connection_idx)
260                        indegree[connection_idx] += 1
261
262        pivots = set()
263        for frontier_idx in frontier:
264            if indegree.get(frontier_idx, 0) == 0:
265                if is_pivot(
266                    frontier_idx,
267                    forest=forest,
268                    threshold=self.pivot_relaxation_steps,
269                ):
270                    pivots.add(frontier_idx)
271
272        return pivots, temp_frontier

274    def base_case(
275        self, upper_bound: int | float, frontier: set[int]
276    ) -> tuple[int | float, set[int]]:
277        """
278        Function:
279
280        - Implements Algorithm 2: Base Case of BMSSP
281
282        Required Arguments:
283        - upper_bound:
284            - Type: int | float
285        - frontier:
286            - Type: set
287            - What: Set with a single vertex x (complete)
288
289        Returns:
290        - new_upper_bound:
291            - Type: int | float
292            - What: The new upper bound for the search
293        - new_frontier:
294            - Type: set[int]
295            - What: Set of vertices v such that distance_matrix[v] < new_upper_bound
296        """
297        assert len(frontier) == 1, "Frontier must be a singleton set"
298        first_frontier = next(iter(frontier))
299
300        new_frontier = set()
301        heap = []
302        heappush(
303            heap,
304            (
305                self.counter_and_edge_distance_matrix[first_frontier],
306                first_frontier,
307            ),
308        )
309        new_upper_bound = upper_bound
310        # Grow until we exceed pivot_relaxation_steps (practical limit), as in Algorithm 2
311        while heap:
312            frontier_distance, frontier_idx = heappop(heap)
313            # Modification: instead of the post process, just break early
314            if len(new_frontier) >= self.pivot_relaxation_steps:
315                new_upper_bound = frontier_distance
316                break
317            new_frontier.add(frontier_idx)
318            prev_distance = self.counter_distance_matrix[frontier_idx]
319            for connection_idx, connection_distance in self.graph[
320                frontier_idx
321            ].items():
322                # Modification:
323                new_distance = (
324                    prev_distance
325                    + connection_distance
326                    + self.counter_value
327                    + self.edge_adj_graph[frontier_idx][connection_idx]
328                )
329                if (
330                    new_distance
331                    <= self.counter_and_edge_distance_matrix[connection_idx]
332                    and new_distance < upper_bound
333                ):
334                    # Addition: Add predecessor tracking
335                    if (
336                        new_distance
337                        < self.counter_and_edge_distance_matrix[connection_idx]
338                    ):
339                        self.predecessor[connection_idx] = frontier_idx
340                        self.counter_and_edge_distance_matrix[
341                            connection_idx
342                        ] = new_distance
343                        self.counter_distance_matrix[connection_idx] = (
344                            prev_distance
345                            + connection_distance
346                            + self.counter_value
347                        )
348                    heappush(heap, (new_distance, connection_idx))
349
350        return new_upper_bound, new_frontier

352    def recursive_bmssp(
353        self, recursion_depth: int, upper_bound: int | float, frontier: set[int]
354    ) -> tuple[int | float, set[int]]:
355        """
356        Function:
357
358        - Implements Algorithm 3: Bounded Multi-Source Shortest Path (BMSSP)
359
360        Required Arguments:
361
362        - recursion_depth:
363            - Type: int
364            - What: The depth of the recursion
365        - upper_bound:
366            - Type: float
367            - What: The upper bound for the search
368        - frontier:
369            - Type: set[int]
370            - What: The set of vertices to explore
371
372        Returns:
373
374        - new_upper_bound:
375            - Type: int | float
376            - What: The new upper bound for the search
377        - new_frontier:
378            - Type: set[int]
379            - What: Set of vertices v such that distance_matrix[v] < new_upper_bound
380        """
381        # Base case
382        if recursion_depth == 0:
383            new_upper_bound, new_frontier = self.base_case(
384                upper_bound, frontier
385            )
386            return new_upper_bound, new_frontier
387
388        # Step 4: Find pivots and temporary frontier
389        pivots, temp_frontier = self.find_pivots(upper_bound, frontier)
390
391        # Step 56: initialize data_struct with pivots
392        # subset_size = 2^((l-1) * t)
393        subset_size = 2 ** ((recursion_depth - 1) * self.target_tree_depth)
394        # Increment the recursion counter for this depth
395        self.recursion_counter[recursion_depth - 1] += 1
396        # Pass the shared recursion data structure map for this depth
397        # Include the current recursion counter as the unique id to ensure
398        # that we don't have stale data in the shared map
399        data_struct = self.data_structure(
400            subset_size=subset_size,
401            upper_bound=upper_bound,
402            recursion_data_id=self.recursion_counter[recursion_depth - 1],
403            recursion_data_list=self.recursion_data_struct_maps[
404                recursion_depth - 1
405            ],
406        )
407        for p in pivots:
408            data_struct.insert_key_value(
409                p, self.counter_and_edge_distance_matrix[p]
410            )
411
412        # Track new_frontier and B' according to Algorithm 3
413        new_frontier = set()
414        # Store the completion_bound for use if the frontier is empty and we break early
415        completion_bound = min(
416            (self.counter_and_edge_distance_matrix[p] for p in pivots),
417            default=upper_bound,
418        )
419
420        # Work budget that scales with level: k*2**(l*t)
421        # k = self.pivot_relaxation_steps
422        # t = self.target_tree_depth
423        work_budget = self.pivot_relaxation_steps * 2 ** (
424            recursion_depth * self.target_tree_depth
425        )
426        # Main loop
427        while len(new_frontier) < work_budget and not data_struct.is_empty():
428            # Step 10: Pull from data_struct: get data_struct_frontier_temp and upper_bound_i
429            data_struct_frontier_bound_temp, data_struct_frontier_temp = (
430                data_struct.pull()
431            )
432
433            # Step 11: Recurse on (l-1, data_struct_frontier_bound_temp, data_struct_frontier_temp)
434            completion_bound, new_frontier_temp = self.recursive_bmssp(
435                recursion_depth - 1,
436                data_struct_frontier_bound_temp,
437                data_struct_frontier_temp,
438            )
439
440            # Track results
441            new_frontier.update(new_frontier_temp)
442
443            # Step 13: Initialize intermediate_frontier to batch-prepend
444            intermediate_frontier = set()
445
446            # Step 1420: relax edges from new_frontier_temp and enqueue into D or intermediate_frontier per their interval
447            for new_frontier_idx in new_frontier_temp:
448                prev_distance = self.counter_distance_matrix[new_frontier_idx]
449                for connection_idx, connection_distance in self.graph[
450                    new_frontier_idx
451                ].items():
452                    new_distance = (
453                        prev_distance
454                        + connection_distance
455                        + self.counter_value
456                        + self.edge_adj_graph[new_frontier_idx][connection_idx]
457                    )
458                    if (
459                        new_distance
460                        <= self.counter_and_edge_distance_matrix[connection_idx]
461                    ):
462                        # Addition: Add predecessor tracking
463                        if (
464                            new_distance
465                            < self.counter_and_edge_distance_matrix[
466                                connection_idx
467                            ]
468                        ):
469                            self.predecessor[connection_idx] = new_frontier_idx
470                            self.counter_and_edge_distance_matrix[
471                                connection_idx
472                            ] = new_distance
473                            self.counter_distance_matrix[connection_idx] = (
474                                prev_distance
475                                + connection_distance
476                                + self.counter_value
477                            )
478                        # Insert based on which interval the new distance falls into
479                        if (
480                            data_struct_frontier_bound_temp
481                            <= new_distance
482                            < upper_bound
483                        ):
484                            data_struct.insert_key_value(
485                                connection_idx, new_distance
486                            )
487                        elif (
488                            completion_bound
489                            <= new_distance
490                            < data_struct_frontier_bound_temp
491                        ):
492                            intermediate_frontier.add(
493                                (connection_idx, new_distance)
494                            )
495
496            # Step 21: Batch prepend intermediate_frontier plus filtered data_struct_frontier_temp in completion_bound, data_struct_frontier_bound_temp)
497            data_struct_frontier_temp_filtered = {
498                (x, self.counter_and_edge_distance_matrix[x])
499                for x in data_struct_frontier_temp
500                if completion_bound
501                <= self.counter_and_edge_distance_matrix[x]
502                < data_struct_frontier_bound_temp
503            }
504
505            data_struct.batch_prepend(
506                intermediate_frontier | data_struct_frontier_temp_filtered
507            )
508        # Optional code if you do not have guaranteed unique lengths.
509        # if len(new_frontier) > work_budget:
510        #     completion_bound = pivot_completion_bound
511
512        # Step 22: Final return
513        completion_bound = min(completion_bound, upper_bound)
514        new_frontier = new_frontier | {
515            v
516            for v in temp_frontier
517            if self.counter_and_edge_distance_matrix[v] < completion_bound
518        }
519
520        return completion_bound, new_frontier

385                upper_bound, frontier
386            )

395        # Increment the recursion counter for this depth
396        self.recursion_counter[recursion_depth - 1] += 1
397        # Pass the shared recursion data structure map for this depth
398        # Include the current recursion counter as the unique id to ensure
399        # that we don't have stale data in the shared map

410                p, self.counter_and_edge_distance_matrix[p]
411            )
412
413        # Track new_frontier and B' according to Algorithm 3
414        new_frontier = set()

445            intermediate_frontier = set()
446

474                            self.counter_distance_matrix[connection_idx] = (
475                                prev_distance
476                                + connection_distance
477                                + self.counter_value
478                            )
479                        # Insert based on which interval the new distance falls into
480                        if (
481                            data_struct_frontier_bound_temp
482                            <= new_distance
483                            < upper_bound
484                        ):
485                            data_struct.insert_key_value(
486                                connection_idx, new_distance
487                            )
488                        elif (
489                            completion_bound
490                            <= new_distance
491                            < data_struct_frontier_bound_temp
492                        ):
493                            intermediate_frontier.add(
494                                (connection_idx, new_distance)
495                            )
496
497            # Step 21: Batch prepend intermediate_frontier plus filtered data_struct_frontier_temp in completion_bound, data_struct_frontier_bound_temp)
498            data_struct_frontier_temp_filtered = {
499                (x, self.counter_and_edge_distance_matrix[x])
500                for x in data_struct_frontier_temp
501                if completion_bound
502                <= self.counter_and_edge_distance_matrix[x]
503                < data_struct_frontier_bound_temp
504            }
505
506            data_struct.batch_prepend(
507                intermediate_frontier | data_struct_frontier_temp_filtered
508            )
509        # Optional code if you do not have guaranteed unique lengths.
510        # if len(new_frontier) > work_budget:
511        #     completion_bound = pivot_completion_bound
512
513        # Step 22: Final return
514        completion_bound = min(completion_bound, upper_bound)
515        new_frontier = new_frontier | {
516            v
517            for v in temp_frontier
518            if self.counter_and_edge_distance_matrix[v] < completion_bound
519        }
520
521        return completion_bound, new_frontier
